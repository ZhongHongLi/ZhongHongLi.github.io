        # 新建gulpfile.js

        -1.gulpfile.js是gulp的配置文件一般在文件的跟目录下

         

















































          四个方法：task,run,src,dest
  
  1.先引入
  const gulp=require('gulp');
  const csso = require('gulp-csso');
  const htmlmin = require('gulp-htmlmin');
  const less = require('gulp-less');
  const fileinclude = require('gulp-file-include');
  const babel = require('gulp-babel'); //es6转es5
  
  
  // html
  gulp.task('htmlmin',()=>{
	  gulp.src('./*.html')
	  .pipe(fileinclude()) //这个方法是合并文件
	  .pipe(htmlmin({collapseWhitespace:true}))// collapseWhitespace 折叠代码已经空格
      .pipe(gulp.dest('dist'));
  
  //命令执行  `gulp htmlmin`
  })
  
  //less 转css 且合并且压缩css
  gulp.task('less', () => {
      // 选择css目录下的所有less文件以及css文件
      gulp.src(['./css/*.less', './src/css/*.css'])
          // 将less语法转换为css语法
          .pipe(less())
          // 将css代码进行压缩
          .pipe(csso())
          // 将处理的结果进行输出
          .pipe(gulp.dest('dist/css'))
		  
		   //命令执行  `gulp less`
  });
 
  //es6 转es5
  gulp.task('jsmin', () => {
      gulp.src('./js/*.js')
          .pipe(babel({
              // 它可以判断当前代码的运行环境 将代码转换为当前运行环境所支持的代码
              presets: ['@babel/env']
          }))
          .pipe(gulp.dest('dist/js'))
		  
		     //命令执行  `gulp jsmin`
  })

        # gulp常用插件

        1. gulp-htmlmin:html文件压缩
        2. gulp-csso:压缩css
        3. gulp-babel:javascript语法转换
        4. gulp-less:less转css
        5. gulp-uglify:压缩混淆js
        6. gulp-file-include公共文件包含
        7. browsersync：浏览器实时更新
